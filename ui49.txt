import { Button } from "./ui/button";
import { Input } from "./ui/input";
import { Textarea } from "./ui/textarea";
import { Label } from "./ui/label";
import { Card } from "./ui/card";
import { Mail, Linkedin, Send, Calendar as CalendarIcon } from "lucide-react";
import { useState } from "react";
import { motion } from "motion/react";
import { useScrollAnimation, staggerContainer, staggerItem } from "./hooks/useScrollAnimation";

export function ContactSection() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    company: '',
    message: ''
  });

  const { ref, isInView } = useScrollAnimation();

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    
    // Create mailto link with form data
    const subject = encodeURIComponent(`New Message from ${formData.name} - ClientFlow Contact Form`);
    const body = encodeURIComponent(`
Name: ${formData.name}
Email: ${formData.email}
Company: ${formData.company}

Message:
${formData.message}

---
This message was sent from the ClientFlow website contact form.
    `);
    
    const mailtoLink = `mailto:kimandersond1@gmail.com?subject=${subject}&body=${body}`;
    window.location.href = mailtoLink;
    
    // Reset form
    setFormData({ name: '', email: '', company: '', message: '' });
  };

  const handleChange = (e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>) => {
    setFormData({
      ...formData,
      [e.target.name]: e.target.value
    });
  };

  return (
    <section id="contact" className="py-20 bg-gradient-to-br from-clientflow-navy to-clientflow-charcoal" ref={ref}>
      <div className="container mx-auto px-6">
        <motion.div 
          className="text-center mb-12"
          initial={{ opacity: 0, y: 30 }}
          animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 30 }}
          transition={{ duration: 0.6, ease: [0.22, 1, 0.36, 1] }}
        >
          <h2 className="text-3xl lg:text-4xl font-bold text-white mb-4">
            Let's Talk Growth
          </h2>
          <div className="w-20 h-1 bg-clientflow-coral mx-auto mb-6"></div>
          <p className="text-xl text-white/90 max-w-2xl mx-auto">
            If you're ready for consistent leads and smoother marketing systems, let's start the conversation.
          </p>
        </motion.div>
        
        <motion.div 
          className="grid lg:grid-cols-2 gap-12 max-w-6xl mx-auto"
          variants={staggerContainer}
          initial="initial"
          animate={isInView ? "animate" : "initial"}
        >
          {/* Contact Form */}
          <motion.div variants={staggerItem}>
            <Card className="p-8 bg-white border-none shadow-xl hover-lift">
              <form onSubmit={handleSubmit} className="space-y-6">
                <motion.div 
                  className="form-focus-glow rounded-lg transition-all duration-300"
                  whileFocus={{ scale: 1.02 }}
                >
                  <Label htmlFor="name" className="text-clientflow-navy">Name *</Label>
                  <Input
                    id="name"
                    name="name"
                    value={formData.name}
                    onChange={handleChange}
                    required
                    className="mt-2 border-gray-200 focus:border-clientflow-coral transition-all duration-300"
                    placeholder="Your full name"
                  />
                </motion.div>
                
                <motion.div 
                  className="form-focus-glow rounded-lg transition-all duration-300"
                  whileFocus={{ scale: 1.02 }}
                >
                  <Label htmlFor="email" className="text-clientflow-navy">Email *</Label>
                  <Input
                    id="email"
                    name="email"
                    type="email"
                    value={formData.email}
                    onChange={handleChange}
                    required
                    className="mt-2 border-gray-200 focus:border-clientflow-coral transition-all duration-300"
                    placeholder="your.email@company.com"
                  />
                </motion.div>
                
                <motion.div 
                  className="form-focus-glow rounded-lg transition-all duration-300"
                  whileFocus={{ scale: 1.02 }}
                >
                  <Label htmlFor="company" className="text-clientflow-navy">Company</Label>
                  <Input
                    id="company"
                    name="company"
                    value={formData.company}
                    onChange={handleChange}
                    className="mt-2 border-gray-200 focus:border-clientflow-coral transition-all duration-300"
                    placeholder="Your company name"
                  />
                </motion.div>
                
                <motion.div 
                  className="form-focus-glow rounded-lg transition-all duration-300"
                  whileFocus={{ scale: 1.02 }}
                >
                  <Label htmlFor="message" className="text-clientflow-navy">Message *</Label>
                  <Textarea
                    id="message"
                    name="message"
                    value={formData.message}
                    onChange={handleChange}
                    required
                    rows={4}
                    className="mt-2 border-gray-200 focus:border-clientflow-coral resize-none transition-all duration-300"
                    placeholder="Tell me about your business and how I can help you grow..."
                  />
                </motion.div>
                
                <motion.div
                  whileHover={{ scale: 1.02 }}
                  whileTap={{ scale: 0.98 }}
                >
                  <Button 
                    type="submit"
                    size="lg"
                    className="w-full bg-clientflow-coral hover:bg-clientflow-coral/90 text-white font-semibold py-3 rounded-full shadow-lg hover:shadow-xl transition-all duration-300 group"
                  >
                    <Send className="mr-2 h-5 w-5 group-hover:translate-x-1 transition-transform" />
                    Start the Conversation
                  </Button>
                </motion.div>
              </form>
            </Card>
          </motion.div>
          
          {/* Contact Information */}
          <motion.div className="space-y-8" variants={staggerItem}>
            <div className="text-white">
              <motion.h3 
                className="text-2xl font-semibold mb-6"
                initial={{ opacity: 0 }}
                animate={isInView ? { opacity: 1 } : { opacity: 0 }}
                transition={{ delay: 0.3, duration: 0.6 }}
              >
                Get in Touch
              </motion.h3>
              <motion.p 
                className="text-lg text-white/90 leading-relaxed mb-8"
                initial={{ opacity: 0 }}
                animate={isInView ? { opacity: 1 } : { opacity: 0 }}
                transition={{ delay: 0.4, duration: 0.6 }}
              >
                Ready to transform your marketing and start attracting quality leads consistently? 
                I'd love to hear about your business goals and discuss how we can achieve them together.
              </motion.p>
            </div>
            
            <div className="space-y-4">
              <motion.a 
                href="mailto:kimandersond1@gmail.com" 
                className="flex items-center space-x-4 p-4 bg-white/10 backdrop-blur-sm rounded-lg hover:bg-white/20 transition-all duration-300 group"
                whileHover={{ x: 4, scale: 1.02 }}
                initial={{ opacity: 0, x: -20 }}
                animate={isInView ? { opacity: 1, x: 0 } : { opacity: 0, x: -20 }}
                transition={{ delay: 0.5, duration: 0.5 }}
              >
                <motion.div 
                  className="w-12 h-12 bg-clientflow-coral rounded-lg flex items-center justify-center"
                  whileHover={{ rotate: 5, scale: 1.1 }}
                  transition={{ duration: 0.3 }}
                >
                  <Mail className="h-6 w-6 text-white" />
                </motion.div>
                <div>
                  <p className="text-white font-medium">Email</p>
                  <p className="text-white/80">kimandersond1@gmail.com</p>
                </div>
              </motion.a>
              
              <motion.a 
                href="https://linkedin.com/in/kim-clientflow" 
                target="_blank" 
                rel="noopener noreferrer"
                className="flex items-center space-x-4 p-4 bg-white/10 backdrop-blur-sm rounded-lg hover:bg-white/20 transition-all duration-300 group"
                whileHover={{ x: 4, scale: 1.02 }}
                initial={{ opacity: 0, x: -20 }}
                animate={isInView ? { opacity: 1, x: 0 } : { opacity: 0, x: -20 }}
                transition={{ delay: 0.6, duration: 0.5 }}
              >
                <motion.div 
                  className="w-12 h-12 bg-clientflow-coral rounded-lg flex items-center justify-center"
                  whileHover={{ rotate: 5, scale: 1.1 }}
                  transition={{ duration: 0.3 }}
                >
                  <Linkedin className="h-6 w-6 text-white" />
                </motion.div>
                <div>
                  <p className="text-white font-medium">LinkedIn</p>
                  <p className="text-white/80">Connect with me professionally</p>
                </div>
              </motion.a>

              <motion.a 
                href="https://calendly.com/kimandersond1" 
                target="_blank" 
                rel="noopener noreferrer"
                className="flex items-center space-x-4 p-4 bg-white/10 backdrop-blur-sm rounded-lg hover:bg-white/20 transition-all duration-300 group animate-pulse-gentle"
                whileHover={{ x: 4, scale: 1.02 }}
                initial={{ opacity: 0, x: -20 }}
                animate={isInView ? { opacity: 1, x: 0 } : { opacity: 0, x: -20 }}
                transition={{ delay: 0.7, duration: 0.5 }}
              >
                <motion.div 
                  className="w-12 h-12 bg-clientflow-coral rounded-lg flex items-center justify-center"
                  whileHover={{ rotate: 5, scale: 1.1 }}
                  transition={{ duration: 0.3 }}
                >
                  <CalendarIcon className="h-6 w-6 text-white" />
                </motion.div>
                <div>
                  <p className="text-white font-medium">Book a Call</p>
                  <p className="text-white/80">Schedule a free consultation</p>
                </div>
              </motion.a>
            </div>
            
            <motion.div 
              className="bg-white/10 backdrop-blur-sm rounded-lg p-6"
              initial={{ opacity: 0, y: 20 }}
              animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 20 }}
              transition={{ delay: 0.8, duration: 0.5 }}
            >
              <h4 className="text-white font-semibold mb-3">Response Time</h4>
              <p className="text-white/90">
                I typically respond to all inquiries within 24 hours during business days. 
                For urgent matters, feel free to mention it in your message.
              </p>
            </motion.div>
          </motion.div>
        </motion.div>
      </div>
    </section>
  );
}
